<?php
// ============================================
// SESSION CHECK - DEBUGGING
// ============================================

// Set session cookie parameters
session_set_cookie_params([
    'lifetime' => 0,
    'path' => '/',
    'domain' => '',
    'secure' => false,
    'httponly' => true,
    'samesite' => 'Lax'
]);

session_start();

echo "<h1>Session Debug Information</h1>";

echo "<h2>Session ID:</h2>";
echo session_id() . "<br>";

echo "<h2>Session Data:</h2>";
echo "<pre>";
print_r($_SESSION);
echo "</pre>";

echo "<h2>Session Status:</h2>";
echo session_status() . " = " . (session_status() == PHP_SESSION_ACTIVE ? "Active" : "Inactive") . "<br>";

echo "<h2>Session Save Path:</h2>";
echo session_save_path() . "<br>";

echo "<h2>Session Cookie Params:</h2>";
echo "<pre>";
print_r(session_get_cookie_params());
echo "</pre>";

echo "<h2>Actions:</h2>";
echo "<a href='?add_test=1'>Add Test Item to Cart</a><br>";
echo "<a href='?clear_test=1'>Clear Cart</a><br>";
echo "<a href='cart.php'>Go to Cart</a><br>";

if (isset($_GET['add_test'])) {
    if (!isset($_SESSION['cart'])) {
        $_SESSION['cart'] = [];
    }
    $_SESSION['cart'][17] = 2;
    $_SESSION['cart'][18] = 1;
    session_write_close();
    echo "<p style='color:green;'>✅ Test items added! Refresh this page.</p>";
}

if (isset($_GET['clear_test'])) {
    $_SESSION['cart'] = [];
    session_write_close();
    echo "<p style='color:red;'>✅ Cart cleared! Refresh this page.</p>";
}
?>
